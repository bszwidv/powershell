-- Active Directory
new-ADOrganizationalUnit BszLehrer -Path "ou=BszUsers,dc=bszlab,dc=edu"


New-ADUser –Name "John Smith" –SamAccountName JohnS –DisplayName "John Smith" –Title "Account Manager" –Enabled $true –ChangePasswordAtLogon $true -AccountPassword (ConvertTo-SecureString "p@ssw0rd" -AsPlainText -force) -PassThru

New-ADGroup -Name "Account Managers" -SamAccountName AcctMgrs -GroupScope Global -GroupCategory Security -Description "Account Managers Group" –PassThru

New-ADOrganizationalUnit -Name AccountsDepartment -ProtectedFromAccidentalDeletion $true  -PassThru

Get-ADUser -Filter { name –like "john*" } ## Gets all the users whose name starts with John

Add-ADGroupMember -Identity AcctMgrs -Members JohnS

Get-ADGroupMember -Identity AcctMgrs

Get-ADPrincipalGroupMembership -Identity JohnS  ## Gets all the groups in which the specified account is a direct member.

Get-ADAccountAuthorizationGroup -Identity JohnS  ## Gets the token groups of an account

Unlock-ADAccount -Identity JohnS

Get-ADDomain -Current LoggedOnUser

Get-ADDomainController -Filter { name -like "*" }  ## Gets all the DCs in the current domain
